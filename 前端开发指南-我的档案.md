```typescript
  baseSalary: number | null;      // 普通员工 null(隐藏)
  bankAccount: string | null;     // 普通员工 null(隐藏)
  bankName: string;

  // 前端交互字段
  editableFields: string[];       // 可编辑字段名集合
  isSensitiveRole: boolean;       // 是否为HR/管理员（控制敏感字段显示）

/** 修改档案请求体（对应接口2） */
export interface MyDetailUpdateRequest {
  employeeId: number;
  currentAddress?: string;
  emergencyContactName?: string;
  emergencyContactPhone?: string;
}

/** 修改日志条目（对应接口3 list item） */
export interface EmployeeChangeLogVO {
  id: number;
  employeeId: number;
  fieldName: string;
  oldValue: string;
  newValue: string;
  changeType: string;
  operatorName: string;

  createTime: string;

export interface PageResult<T> {
  total: number;
  records: T[];
  size: number;
  current: number;
  pages: number;
}

/** 通用响应包装 */

  code: number;

  message: string;
```

## 三、API 接口层

```typescript
// src/pages/Profile/services/profile.ts

import { request } from '@umijs/max';
import type {
  MyProfileVO,
  MyDetailUpdateRequest,
  EmployeeChangeLogVO,
  PageResult,
  BaseResponse,
} from '../types/profile';

const BASE = '/api/employee/profile';

/** 接口1：获取我的档案详情 */
export async function getMyProfile(employeeId?: number) {
  return request<BaseResponse<MyProfileVO>>(`${BASE}/getMyProfile`, {
    method: 'GET',
    params: employeeId ? { employeeId } : {},
  });
}

/** 接口2：修改个人可编辑档案 */
export async function updateMyDetail(data: MyDetailUpdateRequest) {
  return request<BaseResponse<{ employeeId: number; updateTime: string }>>(
    `${BASE}/updateMyDetail`,
    {
      method: 'PUT',
      data,
    },
  );
}

/** 接口3：查询个人档案修改日志（分页） */
export async function getChangeLogs(params: {
  employeeId?: number;
  pageNum: number;
  pageSize: number;
}) {
  return request<BaseResponse<PageResult<EmployeeChangeLogVO>>>(
    `${BASE}/log/list`,
    {
      method: 'GET',
      params,
    },
  );
}
```

### 4.1 页面主入口 `index.tsx`

```tsx
// src/pages/Profile/index.tsx

import React, { useState } from 'react';
import { Tabs, Spin, message } from 'antd';
import { UserOutlined, EditOutlined, HistoryOutlined } from '@ant-design/icons';
import BasicInfoCard from './components/BasicInfoCard';
import DetailInfoCard from './components/DetailInfoCard';
import ChangeLogPanel from './components/ChangeLogPanel';
import EditProfileModal from './components/EditProfileModal';
import { useProfile } from './hooks/useProfile';
import './index.less';

const ProfilePage: React.FC = () => {
  const {
    profile,
    loading,
    refresh,
    updateProfile,
    updating,
  } = useProfile();

  const [editModalVisible, setEditModalVisible] = useState(false);

  const handleSave = async (values: MyDetailUpdateRequest) => {
    const success = await updateProfile(values);
    if (success) {
      message.success('保存成功');
      setEditModalVisible(false);
      refresh(); // 重新加载页面数据
    }
  };

  if (loading) {
    return (
      <div className="profile-loading">
        <Spin size="large" tip="加载中..." />
      </div>
    );
  }

  const tabItems = [
    {
      key: 'overview',
      label: (
        <span><UserOutlined />档案总览</span>
      ),
      children: (
        <div className="profile-overview">
          <BasicInfoCard profile={profile} />
          <DetailInfoCard
            profile={profile}
            onEdit={() => setEditModalVisible(true)}
          />
        </div>
      ),
    },
    {
      key: 'logs',
      label: (
        <span><HistoryOutlined />修改记录</span>
      ),
      children: <ChangeLogPanel employeeId={profile?.id} />,
    },
  ];

  return (
    <div className="profile-page">
      <div className="profile-header">
        <h2>我的档案</h2>
        <span className="profile-subtitle">
          工号：{profile?.employeeNo} | {profile?.statusDesc}
        </span>
      </div>

      <Tabs defaultActiveKey="overview" items={tabItems} />

      <EditProfileModal
        visible={editModalVisible}
        profile={profile}
        loading={updating}
        onSave={handleSave}
        onCancel={() => setEditModalVisible(false)}
      />
    </div>
  );
};

export default ProfilePage;
```

### 4.2 数据 Hook `useProfile.ts`

```typescript
// src/pages/Profile/hooks/useProfile.ts

import { useState, useEffect, useCallback } from 'react';
import { message } from 'antd';
import { getMyProfile, updateMyDetail } from '../services/profile';
import type { MyProfileVO, MyDetailUpdateRequest } from '../types/profile';

export function useProfile(employeeId?: number) {
  const [profile, setProfile] = useState<MyProfileVO | null>(null);
  const [loading, setLoading] = useState(true);
  const [updating, setUpdating] = useState(false);

  const fetchProfile = useCallback(async () => {
    setLoading(true);
    try {
      const res = await getMyProfile(employeeId);
      if (res.code === 0) {
        setProfile(res.data);
      } else {
        message.error(res.message || '获取档案失败');
      }
    } catch (err: any) {
      const msg = err?.response?.data?.message || '网络错误';
      message.error(msg);
      // 401 → 跳转登录
      if (err?.response?.status === 401) {
        window.location.href = '/login';
      }
    } finally {
      setLoading(false);
    }
  }, [employeeId]);

  const updateProfile = useCallback(
    async (values: MyDetailUpdateRequest): Promise<boolean> => {
      setUpdating(true);
      try {
        const res = await updateMyDetail({
          employeeId: profile!.id,
          ...values,
        });
        if (res.code === 0) {
          return true;
        }
        message.error(res.message || '保存失败');
        return false;
      } catch (err: any) {
        message.error(err?.response?.data?.message || '保存失败');
        return false;
      } finally {
        setUpdating(false);
      }
    },
    [profile],
  );

  useEffect(() => {
    fetchProfile();
  }, [fetchProfile]);

  return {
    profile,
    loading,
    updating,
    refresh: fetchProfile,
    updateProfile,
  };
}
```

```tsx
// src/pages/Profile/components/DetailInfoCard.tsx

import React from 'react';

  { key: 'birthday', label: '生日', editable: false },

    label: '直接汇报人',
    editable: false,
  },
  { key: 'workLocation', label: '工作地点', editable: false },
  {
    key: 'contractTypeDesc',
    label: '合同类型',

  },
  { key: 'contractExpireDate', label: '合同到期日', editable: false },
  {
    key: 'probationRatio',

  },

    key: 'bankAccount',

              </span>

        bordered
        size="small"

        style={{ marginTop: 24 }}

  );
};
```

### 4.5 编辑弹窗 `EditProfileModal.tsx`

```tsx
// src/pages/Profile/components/EditProfileModal.tsx

import React, { useEffect } from 'react';
import { Modal, Form, Input, message } from 'antd';
import type { MyProfileVO, MyDetailUpdateRequest } from '../types/profile';

interface Props {
  visible: boolean;
  profile: MyProfileVO | null;
  loading: boolean;
  onSave: (values: MyDetailUpdateRequest) => void;
  onCancel: () => void;
}

const EditProfileModal: React.FC<Props> = ({
  visible,
  profile,
  loading,
  onSave,
  onCancel,
}) => {
  const [form] = Form.useForm();

  // 打开弹窗时回填原值
  useEffect(() => {
    if (visible && profile) {
      form.setFieldsValue({
        currentAddress: profile.currentAddress,
        emergencyContactName: profile.emergencyContactName,
        emergencyContactPhone: profile.emergencyContactPhone,
      });
    }
  }, [visible, profile, form]);

  const handleOk = async () => {
    try {
      const values = await form.validateFields();
      onSave({
        employeeId: profile!.id,
        ...values,
      });
    } catch {
      // 表单校验不通过
    }
  };

  return (
    <Modal
      title="编辑个人档案"
      open={visible}
      onOk={handleOk}
      onCancel={onCancel}
      confirmLoading={loading}
      destroyOnClose
      width={520}
    >
      <Form form={form} layout="vertical" style={{ marginTop: 16 }}>
        <Form.Item
          name="currentAddress"
          label="现居住地址"
          rules={[{ max: 200, message: '地址不超过200字' }]}
        >
          <Input.TextArea
            rows={3}
            placeholder="请输入现居住地址"
            maxLength={200}
            showCount
          />
        </Form.Item>

        <Form.Item
          name="emergencyContactName"
          label="紧急联系人姓名"
          rules={[
            { max: 30, message: '姓名不超过30字' },
          ]}
        >
          <Input placeholder="请输入紧急联系人姓名" maxLength={30} />
        </Form.Item>

        <Form.Item
          name="emergencyContactPhone"
          label="紧急联系人电话"
          rules={[
            {
              pattern: /^1[3-9]\d{9}$/,

          ]}
        >
          <Input placeholder="请输入紧急联系人电话" maxLength={11} />
        </Form.Item>
      </Form>
    </Modal>
  );
};

export default EditProfileModal;
```

### 4.6 修改记录面板 `ChangeLogPanel.tsx`

```tsx
// src/pages/Profile/components/ChangeLogPanel.tsx

import React, { useState, useEffect } from 'react';
import { Table, Tag } from 'antd';
import type { ColumnsType } from 'antd/es/table';
import { getChangeLogs } from '../services/profile';
import type { EmployeeChangeLogVO } from '../types/profile';
import dayjs from 'dayjs';

interface Props {
  employeeId?: number;
}

/** 字段名称中文映射 */
const FIELD_NAME_MAP: Record<string, string> = {
  currentAddress: '现居住地址',
  emergencyContactName: '紧急联系人姓名',
  emergencyContactPhone: '紧急联系人电话',
  employeeName: '姓名',
  gender: '性别',
  email: '邮箱',
};

const ChangeLogPanel: React.FC<Props> = ({ employeeId }) => {
  const [data, setData] = useState<EmployeeChangeLogVO[]>([]);
  const [loading, setLoading] = useState(false);
  const [pagination, setPagination] = useState({ current: 1, pageSize: 10, total: 0 });

  const fetchLogs = async (pageNum: number, pageSize: number) => {
    setLoading(true);
    try {
      const res = await getChangeLogs({
        employeeId,
        pageNum,
        pageSize,
      });
      if (res.code === 0) {
        setData(res.data.records);
        setPagination({
          current: res.data.current,
          pageSize: res.data.size,
          total: res.data.total,
        });
      }
    } finally {
      setLoading(false);
    }
  };

  useEffect(() => {
    fetchLogs(1, 10);
  }, [employeeId]);

  const columns: ColumnsType<EmployeeChangeLogVO> = [
    {
      title: '变更时间',
      dataIndex: 'createTime',
      width: 180,
      render: (val: string) => dayjs(val).format('YYYY-MM-DD HH:mm:ss'),
    },
    {
      title: '变更字段',
      dataIndex: 'fieldName',
      width: 150,
      render: (val: string) => FIELD_NAME_MAP[val] || val,
    },
    {
      title: '原值',
      dataIndex: 'oldValue',
      ellipsis: true,
      render: (val: string) => val || <span style={{ color: '#ccc' }}>（空）</span>,
    },
    {
      title: '新值',
      dataIndex: 'newValue',
      ellipsis: true,
      render: (val: string) => val || <span style={{ color: '#ccc' }}>（空）</span>,
    },
    {
      title: '变更类型',
      dataIndex: 'changeType',
      width: 150,
      render: (val: string) => <Tag color="blue">{val}</Tag>,
    },
    {
      title: '备注',
      dataIndex: 'remark',
      ellipsis: true,
    },
  ];

  return (
    <Table
      columns={columns}
      dataSource={data}
      rowKey="id"
      loading={loading}
      pagination={{
        ...pagination,
        showSizeChanger: true,
        showTotal: (total) => `共 ${total} 条记录`,
        onChange: (page, size) => fetchLogs(page, size),
      }}
    />
  );
};

export default ChangeLogPanel;
```

## 六、页面完整业务流程（前端调用顺序）

```
┌─────────────────────────────────────────────────────┐
│ 1. 进入「我的档案」页面                                │
│    调用 GET /api/employee/profile/getMyProfile      │
│    → 渲染 用工基础信息（置灰） + 个人详细档案           │
└──────────────────────┬──────────────────────────────┘
                       │
                       ▼
┌─────────────────────────────────────────────────────┐
│ 2. 字段状态区分                                      │
│    全部字段 → 置灰不可编辑                              │
│    仅 currentAddress / emergencyContactName          │
│         / emergencyContactPhone 显示蓝色可编辑标记     │
│    薪资/银行卡 → 普通员工显示"****（仅HR可见）"          │
└──────────────────────┬──────────────────────────────┘
                       │
                       ▼ (点击"编辑档案"按钮)
┌─────────────────────────────────────────────────────┐
│ 3. 弹出 EditProfileModal                             │
│    → 回填接口1返回的 currentAddress                    │
│               emergencyContactName                   │
│               emergencyContactPhone                  │
└──────────────────────┬──────────────────────────────┘
```

## 七、异常处理清单

| 场景 | HTTP状态码 | 后端ErrorCode | 前端处理 |
|---|---|---|---|
| Token过期/未登录 | 401 | 40100 (NOT_LOGIN) | 拦截器统一跳转 `/login` |
| 已离职员工修改 | 200 | 50026 (EMPLOYEE_RESIGNED) | message.error 提示 |
| 越权查看他人档案 | 200 | 50027 (PERMISSION_DENIED) | message.error 提示 |
| 越权修改他人档案 | 200 | 50028 (UPDATE_DENIED) | message.error 提示 |
| 手机号格式错误 | 200 | 50029 (PHONE_FORMAT_ERROR) | 表单校验拦截 + message.error |
| 网络异常 | - | - | message.error("网络异常，请稍后重试") |

## 八、权限矩阵（前端控制）

| 角色 | 查自己 | 查他人 | 脱敏等级 | 可编辑 | 查看日志 |
|---|---|---|---|---|---|
| 普通员工 | ✅ | ❌ | 手机/身份证脱敏，薪资/银行卡隐藏 | ✅（仅3字段） | ✅（仅自己） |
| 管理员/HR | ✅ | ✅（传 employeeId） | 全部明文 | ✅（仅3字段） | ✅（所有人） |

前端判断逻辑：
```typescript
// 判断是否可编辑某个字段
const isEditable = (fieldKey: string) =>
  profile?.editableFields?.includes(fieldKey);

// 判断是否显示完整薪资
const showFullSalary = profile?.isSensitiveRole;  // HR/管理员
```
